================================================================================
WHAT THE SCRIPT DOES
================================================================================

```bash
#!/bin/bash
# auto_test_keys.sh - Fully automatic key tester

# All 832 keys
keys=(
    "000000" "017C0A" "017C0a" "017c0A" "017c0a" "0XUEKN" "0XUEKn"
    ... (all 832 keys) ...
    "xnkjUM" "xnkjUm" "xnkjuM" "xnkjum"
)

# Screen coordinates (OPPO K13 5G - 1080 x 2400)
ENTRY_KEY="540 943"
KEY_FIELD="540 541"
ACTIVATE="834 657"
CLOSE="474 608"

# Loop through all keys
for i in "${!keys[@]}"; do
    key="${keys[$i]}"
    echo "Testing key $((i+1))/832: $key"

    # Open SE app
    adb shell am start -n com.snake/.MainActivity
    sleep 2

    # Tap Entry Key
    adb shell input tap $ENTRY_KEY
    sleep 2

    # Tap key field
    adb shell input tap $KEY_FIELD
    sleep 1

    # Type the key
    adb shell input text "$key"
    sleep 1

    # Tap Activate
    adb shell input tap $ACTIVATE
    sleep 5

    # Take screenshot
    adb shell screencap -p "/sdcard/Pictures/KeyTests/key_$(printf '%03d' $((i+1)))_${key}.png"
    echo "Screenshot saved"

    # Check if valid (you can add OCR here)
    # For now, just tap Close and continue

    # Tap Close button
    adb shell input tap $CLOSE
    sleep 1

    # Close app
    adb shell am force-stop com.snake
    sleep 1

    # Progress notification every 50 keys
    if [ $((i % 50)) -eq 0 ]; then
        echo "Progress: $((i+1))/832 keys tested"
    fi
done

echo "Testing complete! Check /sdcard/Pictures/KeyTests/ for screenshots"
```

================================================================================
TROUBLESHOOTING
================================================================================

ADB not connecting?
→ Run: adb kill-server
→ Run: adb tcpip 5555
→ Run: adb connect localhost:5555

Script not executable?
→ Run: chmod +x auto_test_keys.sh

Taps missing buttons?
→ Edit coordinates in script
→ Adjust ±10 pixels if needed

Screen turns off?
→ Keep phone plugged in
→ Or disable auto-sleep in settings
